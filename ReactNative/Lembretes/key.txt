As keys normalmente são utilizadas buscando o id do banco de dados, porém no exemplo do exercicio, foi utilizado o mesmo name, pois ele tb serve de referencia.
Unico detalhe é que cada elemento da list deve ser unico, não pode ser passado dois com o mesmo valor

//Ex.
export default class App extends React.Component {
  renderList() {
    const names = [
      'Van Hallen',
      'Jimmy Hendrix',
      'Steve Vai',
      'Joe Satriani',
      'Steve Ray Vaughan'
    ];

    // array1.map(callbackfn[, thisArg])
    
    const textElements = names.map(name => {
      return <Text key={name}>{name}</Text>
    })

    return textElements;
  }
